    slides: any;

        this.slides.changes
            .subscribe((slide: any) => {
                this.slideWidth = slide._results[0].nativeElement.offsetWidth;
                this.marginLeft = +parseFloat(getComputedStyle(slide._results[1] && slide._results[1].nativeElement).marginLeft);
                this.slidesLength = slide._results.length;
            });
